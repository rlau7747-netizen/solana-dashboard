    <script>
        async function refreshData() {
            try {
                const response = await fetch('/data');
                const data = await response.json();
                document.getElementById('portfolioValue').innerText = `$${data.portfolioValue.toFixed(2)}`;
                document.getElementById('profit24h').innerText = `+$${data.profit24h.toFixed(2)}`;
                document.getElementById('return24h').innerText = `+${data.return24h}%`;
                document.getElementById('activeTrades').innerText = `${data.activeTrades[0].pair} +$${data.activeTrades[0].profit} ${data.activeTrades[0].status}`;
                document.getElementById('watchlist').innerText = data.watchlist.length ? data.watchlist.join(', ') : 'No pairs in watchlist';
                document.getElementById('opportunity').innerText = `Pair: ${data.currentOpportunity.pair}, Profit: $${data.currentOpportunity.profit} (${data.currentOpportunity.profitPct}%), Buy: $${data.currentOpportunity.buyPrice}, Sell: $${data.currentOpportunity.sellPrice}`;

                const table = document.getElementById('recentTable');
                table.innerHTML = '<tr><th>Pair</th><th>Time</th><th>Profit</th><th>Status</th></tr>';  // Reset table
                data.recentExecutions.forEach((trade) => {
                    const row = `<tr><td>${trade.pair}</td><td>${trade.time}</td><td class="green">+$${trade.profit}</td><td>${trade.status}</td></tr>`;
                    table.innerHTML += row;
                });
            } catch (error) {
                document.getElementById('opportunity').innerText = 'Error loading data';
            }
        }

        refreshData();  // Load awal
        setInterval(refreshData, 10000);  // Auto-refresh setiap 10 detik
    </script>
